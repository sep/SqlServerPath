## Usage

    var sqlDataPathRoot = SqlServerPath.Configuration.GetSqlDataRoot;
    var drive = sqlDataPathRoot.drive;  // e.g. C
    var path = sqlDataPathRoot.path;    // e.g. Path/To/DataFiles/DATA
